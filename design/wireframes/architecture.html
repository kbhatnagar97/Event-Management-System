Z stores/useCheckInStore.ts TypeScript 
import { create } from 'zustand';
import type { Guest, OverlayType } from '@/types/checkin';

interface CheckInState {
  screen: 'pin' | 'scanner' | 'search';
  overlay: OverlayType | null;
  overlayStep: 1 | 2;
  currentGuest: Guest | null;
  editAdults: number;
  editKids: number;
  searchQuery: string;
  searchFilter: 'all' | 'pending' | 'checked_in';
  codeEntryOpen: boolean;
  enteredCode: string;

  // Actions
  setScreen: (s: CheckInState['screen']) => void;
  showOverlay: (type: OverlayType, guest: Guest) => void;
  advanceOverlay: () => void;
  dismissOverlay: () => void;
  updateEditCounter: (type: 'adults' | 'kids', delta: number) => void;
  setSearchQuery: (q: string) => void;
  setSearchFilter: (f: CheckInState['searchFilter']) => void;
  toggleCodeEntry: () => void;
  setEnteredCode: (c: string) => void;
}

export const useCheckInStore = create<CheckInState>((set) => ({
  screen: 'pin',
  overlay: null,
  overlayStep: 1,
  currentGuest: null,
  editAdults: 0,
  editKids: 0,
  searchQuery: '',
  searchFilter: 'all',
  codeEntryOpen: false,
  enteredCode: '',

  setScreen: (s) => set({ screen: s }),

  showOverlay: (type, guest) =>
    set({
      overlay: type,
      overlayStep: 1,
      currentGuest: guest,
      editAdults: guest.adults,
      editKids: guest.kids,
    }),

  advanceOverlay: () =>
    set({ overlayStep: 2 }),

  dismissOverlay: () =>
    set({
      overlay: null,
      overlayStep: 1,
      currentGuest: null,
    }),

  updateEditCounter: (type, delta) =>
    set((state) => ({
      [type === 'adults' ? 'editAdults' : 'editKids']:
        Math.max(0, (type === 'adults' ? state.editAdults : state.editKids) + delta),
    })),

  setSearchQuery: (q) => set({ searchQuery: q }),
  setSearchFilter: (f) => set({ searchFilter: f }),
  toggleCodeEntry: () =>
    set((state) => ({ codeEntryOpen: !state.codeEntryOpen })),
  setEnteredCode: (c) => set({ enteredCode: c }),
}));
H hooks/useCheckIn.ts TypeScript 
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { checkInGuest, undoCheckIn } from '@/api/checkin';
import { useCheckInStore } from '@/stores/useCheckInStore';
import type { CheckInPayload } from '@/types/checkin';

export function useCheckIn() {
  const queryClient = useQueryClient();
  const { advanceOverlay, dismissOverlay } = useCheckInStore();

  const checkIn = useMutation({
    mutationFn: (payload: CheckInPayload) =>
      checkInGuest(payload),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['guests'] });
      queryClient.invalidateQueries({ queryKey: ['stats'] });
      advanceOverlay();
      setTimeout(dismissOverlay, 3000);
    },
  });

  const undo = useMutation({
    mutationFn: (guestId: string) =>
      undoCheckIn(guestId),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['guests'] });
      queryClient.invalidateQueries({ queryKey: ['stats'] });
      advanceOverlay();
      setTimeout(dismissOverlay, 3000);
    },
  });

  return { checkIn, undo };
}
R features/checkin/components/SuccessOverlay/index.tsx TSX 
import { useCheckInStore } from '@/stores/useCheckInStore';
import { useCheckIn } from '@/hooks/useCheckIn';
import { AnimatedIcon } from '@/components/AnimatedIcon';
import './index.scss';

export function SuccessOverlay() {
  const {
    overlay, overlayStep, currentGuest,
    editAdults, editKids,
    updateEditCounter, dismissOverlay,
  } = useCheckInStore();

  const { checkIn } = useCheckIn();

  if (overlay !== 'success' || !currentGuest) return null;

  const handleConfirm = () => {
    checkIn.mutate({
      guestId: currentGuest.id,
      adults: editAdults,
      kids: editKids,
      method: 'qr_scan',
    });
  };

  return (
    <div className="overlay">
      <div className="overlay-backdrop" />
      <div className="overlay-card">
        {overlayStep === 1 ? (
          <>
            <AnimatedIcon type="success" />
            <h2>Guest Found</h2>
            {/* ... verify details + counters ... */}
            <button onClick={handleConfirm}>
              Confirm Check-In
            </button>
          </>
        ) : (
          <>
            <AnimatedIcon type="celebrate" />
            <h2>Checked In!</h2>
            {/* ... auto-dismiss countdown ... */}
          </>
        )}
      </div>
    </div>
  );
}
